CREATE DATABASE PROJETOLOCACAO;

USE PROJETOLOCACAO;

CREATE TABLE CLIENTE(
NOME VARCHAR(30),
CPF VARCHAR(14),
SEXO CHAR(1),
DATANASCIMENTO DATE,
RG VARCHAR(12),
NACIONALIDADE VARCHAR(30),
TELEFONE VARCHAR(30)
);

CREATE TABLE VEICULOS(
MARCA VARCHAR(30),
MODELO VARCHAR(30),
ANO INT(4),
PLACA VARCHAR(12),
COR VARCHAR(30)
);
